# efinmap cron jobs — install with: crontab -u www-data deploy/crontab.txt
#
# Prerequisite (run once as root):
#   mkdir -p /var/log/efinmap
#   chown www-data:www-data /var/log/efinmap
#
# All collectors must run from /opt/efinmap/collectors so flat imports
# (from proxy import ..., from db import ...) resolve. proxy.py loads
# /opt/efinmap/.env via Path(__file__).parent.parent regardless of cwd.

PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin
PYTHON=/opt/efinmap/venv/bin/python
COLLECTORS=/opt/efinmap/collectors
LOG=/var/log/efinmap

# US market — every 15 min
*/15 * * * * cd $COLLECTORS && $PYTHON collect_us.py   >> $LOG/us.log   2>&1

# HK market — every 15 min
*/15 * * * * cd $COLLECTORS && $PYTHON collect_hk.py   >> $LOG/hk.log   2>&1

# Asia (JP/KR/TW) — every 20 min
*/20 * * * * cd $COLLECTORS && $PYTHON collect_asia.py >> $LOG/asia.log 2>&1

# Snapshot cleanup — daily at 03:30 server time, keeps last 30 days
30 3 * * * cd $COLLECTORS && $PYTHON cleanup_snapshots.py >> $LOG/cleanup.log 2>&1
